async function getDadAISuggestions() { const focus = document.getElementById("dadFocusInput").value.trim(); if (!focus) { alert("Please enter your weekly focus first"); return; } document.getElementById("dadAiLoading").style.display = "block"; document.getElementById("dadAiBox").style.display = "none"; try { const res = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "text/plain" }, body: JSON.stringify({ action: "aiDadSuggest", goal: focus }) }); const data = await res.json(); document.getElementById("dadAiLoading").style.display = "none"; if (!data.success || !data.steps) { document.getElementById("dadAiBox").style.display = "block"; document.getElementById("dadAiBox").innerHTML = "Couldn't load suggestions. Please try again."; return; } document.getElementById("dadAiBox").style.display = "block"; document.getElementById("dadAiBox").innerHTML = "✨ Suggested steps:

" + data.steps.map((s, i) => `${i+1}. ${s}`).join("
"); } catch(e) { document.getElementById("dadAiLoading").style.display = "none"; } }
